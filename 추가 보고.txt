    function solution(s) {
  let result = [];
  let i = 0;
  while (i < s.length) {
    let x = s[i];
    let countX = 1;
    let countOther = 0;
    for (let j = i + 1; j < s.length; j++) {
      if (s[j] === x) {
        countX++;
      } else {
        countOther++;
      }
      if (countX === countOther) {
        result.push(s.slice(i, j + 1));
        i = j + 1;
        break;
      }
      if (j === s.length - 1) {
        result.push(s.slice(i));
        i = s.length;
      }
    }
  }
  return result.length;
}
